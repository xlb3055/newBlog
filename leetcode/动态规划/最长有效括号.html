  3. жңҖз»ҲжүҖжңүиҝһз»ӯзҡ„ true ж®өиҗҪпјҢе°ұжҳҜжңүж•ҲжӢ¬еҸ·еҢәй—ҙгҖӮ

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // иҜ»еҸ–еӯ—з¬ҰдёІ
        String s = scanner.nextLine();

        Solution solution = new Solution();
        int result = solution.longestValidParentheses(s);

        System.out.println(result);

        scanner.close();
    }
}

class Solution {
    public int longestValidParentheses(String s) {
        int n = s.length();
        Stack<Integer> stack = new Stack<>();
        boolean[] dp = new boolean[n];  // dp[i] = true иЎЁзӨәиҜҘдҪҚзҪ®еңЁжңүж•ҲжӢ¬еҸ·дёӯ

        for (int i = 0; i < n; i++) {
            char c = s.charAt(i);
            if (c == '(') {
                stack.push(i);
            } else {
                if (!stack.isEmpty()) {
                    int index = stack.pop();
                    dp[index] = true;
                    dp[i] = true;
                }
            }
        }

        // жүҫжңҖй•ҝиҝһз»ӯзҡ„ true зҡ„еҢәй—ҙ
        int max = 0, cur = 0;
        for (int i = 0; i < n; i++) {
            if (dp[i]) {
                cur++;
                max = Math.max(max, cur);
            } else {
                cur = 0;
            }
        }

        return max;
    }
}
