438. жүҫеҲ°еӯ—з¬ҰдёІдёӯжүҖжңүеӯ—жҜҚејӮдҪҚиҜҚ

йўҳзӣ®пјҡжүҫеҲ°еӯ—з¬ҰдёІдёӯжүҖжңүеӯ—жҜҚејӮдҪҚиҜҚ

жҸҸиҝ°пјҡ

з»ҷе®ҡдёӨдёӘеӯ—з¬ҰдёІ s е’Ң pпјҢжүҫеҲ°еӯ—з¬ҰдёІ s дёӯжүҖжңү p зҡ„ејӮдҪҚиҜҚзҡ„еӯҗдёІпјҢиҝ”еӣһиҝҷдәӣеӯҗдёІзҡ„иө·е§Ӣзҙўеј•гҖӮдёҚиҖғиҷ‘зӯ”жЎҲиҫ“еҮәзҡ„йЎәеәҸгҖӮ

зӨәдҫӢпјҡ

иҫ“еҮә: [0, 6]
и§ЈйҮҠ:
иө·е§Ӣзҙўеј• 0 зҡ„еӯҗдёІжҳҜ "cba"пјҢжҳҜ "abc" зҡ„ејӮдҪҚиҜҚгҖӮ

зӨәдҫӢ 2:

иҫ“е…Ҙ: s = "abab", p = "ab"
иҫ“еҮә: [0, 1, 2]
и§ЈйҮҠ:
иө·е§Ӣзҙўеј• 0 зҡ„еӯҗдёІжҳҜ "ab"пјҢжҳҜ "ab" зҡ„ејӮдҪҚиҜҚгҖӮ
иө·е§Ӣзҙўеј• 1 зҡ„еӯҗдёІжҳҜ "ba"пјҢжҳҜ "ab" зҡ„ејӮдҪҚиҜҚгҖӮ
иө·е§Ӣзҙўеј• 2 зҡ„еӯҗдёІжҳҜ "ab"пјҢжҳҜ "ab" зҡ„ејӮдҪҚиҜҚгҖӮ

ж ёеҝғжҖқжғіпјҡж»‘еҠЁзӘ—еҸЈ

  1. ејӮдҪҚиҜҚзҡ„зү№жҖ§пјҡ

    • дёӨдёӘеӯ—з¬ҰдёІжҳҜејӮдҪҚиҜҚж„Ҹе‘ізқҖе®ғд»¬зҡ„еӯ—з¬Ұж•°йҮҸе®Ңе…ЁдёҖиҮҙгҖӮжҚўеҸҘиҜқиҜҙпјҢдёӨдёӘеӯ—з¬ҰдёІзҡ„еӯ—з¬Ұйў‘зҺҮеҲҶеёғжҳҜзӣёеҗҢзҡ„гҖӮ

  2. ж»‘еҠЁзӘ—еҸЈпјҡ

    • дҪҝз”Ёж»‘еҠЁзӘ—еҸЈеңЁеӯ—з¬ҰдёІ s дёҠжһ„йҖ дёҖдёӘй•ҝеәҰдёә p.length зҡ„зӘ—еҸЈгҖӮ
    • з»ҙжҠӨзӘ—еҸЈеҶ…зҡ„еӯ—з¬Ұйў‘зҺҮпјҢдёҺзӣ®ж Үеӯ—з¬ҰдёІ p зҡ„еӯ—з¬Ұйў‘зҺҮиҝӣиЎҢжҜ”иҫғгҖӮ

  3. е®һзҺ°жӯҘйӘӨпјҡ

    1. жһ„йҖ зӣ®ж Үеӯ—з¬ҰдёІ p зҡ„еӯ—з¬Ұйў‘зҺҮз»ҹи®Ўж•°з»„гҖӮ
    2. еңЁеӯ—з¬ҰдёІ s дёҠж»‘еҠЁзӘ—еҸЈпјҢжҜҸж¬Ўеҗ‘еҸіз§»еҠЁдёҖдёӘеӯ—з¬ҰпјҢжӣҙж–°еҪ“еүҚзӘ—еҸЈзҡ„еӯ—з¬Ұйў‘зҺҮз»ҹи®Ўж•°з»„гҖӮ
    3. жҜ”иҫғеҪ“еүҚзӘ—еҸЈзҡ„йў‘зҺҮж•°з»„е’Ңзӣ®ж Үйў‘зҺҮж•°з»„пјҢеҰӮжһңзӣёеҗҢпјҢеҲҷи®°еҪ•иө·е§Ӣзҙўеј•гҖӮ
    4. йҮҚеӨҚдёҠиҝ°иҝҮзЁӢзӣҙеҲ°ж»‘еҠЁзӘ—еҸЈйҒҚеҺҶе®Ңж•ҙдёӘеӯ—з¬ҰдёІгҖӮ

д»Јз Ғе®һзҺ°

import java.util.Scanner;
import java.util.ArrayList;
import java.util.List;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // Read strings s and p
        String s = scanner.nextLine();
        String p = scanner.nextLine();

        Solution solution = new Solution();
        List<Integer> result = solution.findAnagrams(s, p);

        // Print result list
        System.out.println(result);
        scanner.close();
    }
}

class Solution {
    public List<Integer> findAnagrams(String s, String p) {
        List<Integer> result = new ArrayList<>();
        if (s.length() < p.length()) {
            return result;
        }

        // еӯ—з¬Ұйў‘зҺҮж•°з»„
        int[] pCount = new int[26];
        int[] sCount = new int[26];

        // з»ҹи®Ўзӣ®ж Үеӯ—з¬ҰдёІ p зҡ„еӯ—з¬Ұйў‘зҺҮ
        for (char c : p.toCharArray()) {
            pCount[c - 'a']++;
        }

        // еҲқе§ӢеҢ–зӘ—еҸЈзҡ„еӯ—з¬Ұйў‘зҺҮ
        for (int i = 0; i < p.length(); i++) {
            sCount[s.charAt(i) - 'a']++;
        }

        // ж»‘еҠЁзӘ—еҸЈйҒҚеҺҶеӯ—з¬ҰдёІ s
        for (int i = 0; i <= s.length() - p.length(); i++) {
            // жҜ”иҫғеҪ“еүҚзӘ—еҸЈе’Ңзӣ®ж Үйў‘зҺҮж•°з»„
            if (isAnagram(sCount, pCount)) {
                result.add(i); // и®°еҪ•иө·е§Ӣзҙўеј•
            }

            // ж»‘еҠЁзӘ—еҸЈеҗ‘еҸіз§»еҠЁ
            if (i + p.length() < s.length()) {
                sCount[s.charAt(i) - 'a']--; // з§»йҷӨзӘ—еҸЈжңҖе·Ұдҫ§еӯ—з¬Ұ
                sCount[s.charAt(i + p.length()) - 'a']++; // ж·»еҠ зӘ—еҸЈжңҖеҸідҫ§еӯ—з¬Ұ
            }
        }

        return result;
    }

    // еҲӨж–ӯдёӨдёӘеӯ—з¬Ұйў‘зҺҮж•°з»„жҳҜеҗҰзӣёзӯү
    private boolean isAnagram(int[] sCount, int[] pCount) {
        for (int i = 0; i < 26; i++) {
            if (sCount[i] != pCount[i]) {
                return false;
            }
        }
        return true;
    }
}
